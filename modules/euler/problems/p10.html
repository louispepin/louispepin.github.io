Problem 10: Summation of primes

The sum of the primes below 10 is 2 + 3 + 5 + 7 = 17.
Find the sum of all the primes below two million. 


    var isPrime = function (x) {
        var prime = true;
        for (var i = 2; i * i <= x; i++) {
            if (x % i == 0)
                prime = false;
        }

        return prime;
    };

    // start at 3, skip even numbers
    var sum = 2;
    for (var i = 3; i < 2000000; i+=2) {
        if (isPrime(i))
            sum += i;
    }

    return sum;
