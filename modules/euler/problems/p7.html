Problem 7: 10001st prime

By listing the first six prime numbers: 2, 3, 5, 7, 11, and 13, we can see that the 6th prime is 13.
What is the 10 001st prime number? 


    var count = 1;
    var num = 2;
    var primes = [2];

    // taken from problem 3
    var isPrime = function (x) {
        var prime = true;
        for (var i = 2; i * i <= x; i++) {
            if (x % i == 0)
                prime = false;
        }

        return prime;
    };

    // brute force, skip even numbers
    while (count < 10001) {
        num++;

        if (num % 2 == 0)
            continue;

        if (isPrime(num)) {
            count++;
            primes.push(num);
        }
    }

    return num;
